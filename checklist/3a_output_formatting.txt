def emotion_detector(text_to_analyze: str):
    """Perform post request to NLP library to analyze the emotions of text -> text_to_analyze.
       Then return the result of analyzis as a dictionary."""
    URL = 'https://sn-watson-emotion.labs.skills.network/v1/watson.runtime.nlp.v1/NlpService/EmotionPredict'
    headers = {"grpc-metadata-mm-model-id": "emotion_aggregated-workflow_lang_en_stock"}
    input_json = {'raw_document': {'text': text_to_analyze}}
    response = requests.post(url=URL, headers=headers, json=input_json)
    res_json = json.loads(response.text)
    res_trimmed = res_json['emotionPredictions'][0]['emotion']
    emotion_score = {}
    emotion_score['anger'] = res_trimmed['anger']
    emotion_score['disgust'] = res_trimmed['disgust']
    emotion_score['fear'] = res_trimmed['fear']
    emotion_score['joy'] = res_trimmed['joy']
    emotion_score['sadness'] = res_trimmed['sadness']
    dominant_emotion = max(emotion_score, key=emotion_score.get)
    return {'anger': emotion_score['anger'], 
            'disgust': emotion_score['disgust'],
            'fear': emotion_score['fear'],
            'joy': emotion_score['joy'],
            'sadness': emotion_score['sadness'],
            'dominant_emotion': dominant_emotion }
