"""This file has all methods related to web application and its routes"""
from flask import Flask, request, render_template
from EmotionDetection.emotion_detection import emotion_detector

app = Flask(__name__)

@app.route('/emotionDetector')
def emotion_analysis():
    """This method acts takes all requests for the route /emotionDetector"""
    text_to_analyze = request.args.get('textToAnalyze')
    res = emotion_detector(text_to_analyze)
    if res['dominant_emotion'] is None:
        text = "Invalid text! Please try again!."
    else:
        text = "For the given statement, the system response is " \
                + f"'anger': {res['anger']}, 'disgust': {res['disgust']}, " \
                + f"'fear': {res['fear']}, 'joy': {res['joy']} and " \
                + f"'sadness': {res['sadness']}. The dominant emotion is {res['dominant_emotion']}."
    return text

@app.route('/')
def index_load_method():
    """Method for default route /"""
    return render_template('index.html')

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000)
